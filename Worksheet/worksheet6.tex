\begin{problem}
    \begin{enumerate}
        \item Prove that $\Q\subseteq \R$.
        \item Prove that $\R \not\subseteq \Q$.
    \end{enumerate}
\end{problem}

\begin{problem}
    Prove that, 
   \begin{equation*}
   \set[\big]{ x \in \R \st x^2 \leq 1 } = [-1,1] \,.
   \end{equation*}
\end{problem}

\begin{problem}
    Prove that for any two sets $X$ and $Y$, 
    \begin{equation*}
        X\cap Y \subseteq X \cup Y \,.
    \end{equation*}
\end{problem}
