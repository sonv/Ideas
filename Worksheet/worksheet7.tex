\newpage


\begin{question}
	Prove by induction that
	\begin{equation*}
		1+2+\dots + n = \frac{n(n+1)}{2} \,.
	\end{equation*}
\end{question}

\vspace{8cm}

\begin{question}
	Let $S_n$ be the sum of the first $n$ natural numbers.
	Show by induction that for any $n\in \N$,
	\begin{equation*}
		S_n + S_{n+1} = (n+1)^2 \,.
	\end{equation*}
\end{question}

\vspace{8cm}

\begin{question}
	Show by induction that
	\begin{equation*}
		1^2 + 2^2 + \dots + n^2 = \frac{n(n+1)(2n+1)}{6} \,.
	\end{equation*}
\end{question}
